# 3.6 技能动画

## 1. 动画系统概述

无名杀的动画系统仅有动效，而无实际效果，总计包括:
- 技能发动动画
- 特效动画
- 卡牌动画
- 状态动画

## 2. 技能发动动画

### 2.1 基础动画
```javascript
"animation_skill": {
    // 技能动画
    skillAnimation: true,           // 开启技能动画
    animationStr: "技能发动",      // 动画文字
    animationColor: "fire",        // 动画颜色(fire/thunder/water/metal/soil)
    
    content: function(){
        // 技能效果
    }
}
```

### 2.2 自定义动画
```javascript
"custom_animation": {
    content: async function (event, trigger, player){
        // 播放指定技能动画
        player.$skill('技能名');
        
        // 自定义动画效果
        player.$fire();            // 火焰效果
        player.$thunder();         // 雷电效果
        player.$fullscreenpop('文字', 'fire'); // 全屏特效
    }
}
```

## 3. 特效动画

### 3.1 基础特效
```javascript
content: async function (event, trigger, player){
    // 基础特效
    player.$gain(cards);          // 获得牌动画
    player.$give(num, target);    // 给牌动画
    player.$throw(cards);         // 弃牌动画
    
    // 状态特效
    player.$damage('fire');       // 受伤特效
    player.$recover();            // 回复特效
    player.$shield();             // 护盾特效
}
```

### 3.2 高级特效
```javascript
"effect_skill": {
    content: async function (event, trigger, player){
        // 连续特效
        player.$fire();
        await game.delay(0.5);
        player.$thunder();
        
        // 多目标特效
        for(let target of targets){
            target.$damage('fire');
            await game.delay(0.2);
        }
    }
}
```

## 4. 卡牌动画

### 4.1 使用动画
```javascript
"card_animation": {
    content: async function (event, trigger, player){
        // 卡牌使用动画
        player.$useCard(card, targets);
        
        // 卡牌打出动画
        player.$throw(card);
        
        // 卡牌获得动画
        player.$draw(num);
        player.$gain(cards);
    }
}
```

### 4.2 特殊动画
```javascript
"special_card": {
    content: async function (event, trigger, player){
        // 判定动画
        player.$judge(card);
        
        // 展示动画
        player.$showCards(cards);
        
        // 比较动画
        player.$compare(card1, target, card2);
    }
}
```

## 5. 状态动画

### 5.1 基础状态
```javascript
"state_animation": {
    content: async function (event, trigger, player){
        // 翻面动画
        player.$turnOver();
        
        // 横置动画
        player.$link();
        
        // 濒死动画
        player.$die();
    }
}
```

### 5.2 标记动画
```javascript
"mark_animation": {
    content: async function (event, trigger, player){
        // 添加标记动画
        player.$mark('标记名', {
            name: '标记名称',
            content: '标记描述'
        });
        
        // 移除标记动画
        player.$removeMark('标记名');
    }
}
```

## 6. 进阶技巧

### 6.1 动画序列
```javascript
"sequence_animation": {
    content: async function (event, trigger, player){
        // 创建动画序列
        await player.$fire();
        await game.delay(0.5);
        await player.$thunder();
        await game.delay(0.3);
        await player.$fullscreenpop('技能发动', 'fire');
    }
}
```

### 6.2 条件动画
```javascript
"condition_animation": {
    content: async function (event, trigger, player){
        // 根据条件播放不同动画
        if(player.hp < 3){
            await player.$fire();
        } else {
            await player.$thunder();
        }
        
        // 动态特效
        let type = player.hp < 3 ? 'fire' : 'thunder';
        await player.$damage(type);
    }
}
```

## 7. 注意事项

1. **性能优化**
   - 避免过多动画
   - 控制动画时长
   - 减少重复动画

1. 创建一个基础动画技能：
   - 添加技能发动动画
   - 包含基础特效
   - 设置合适的延时

<details>
<summary>参考答案 | 🟩 Easy</summary>

```javascript
"basic_animation": {
    enable: "phaseUse",
    usable: 1,
    // 技能动画
    skillAnimation: true,
    content: async function (event, trigger, player){
        // 播放技能动画
        player.$skill('basic_animation',"legend","fire");
        game.delay(0.5);

        // 基础特效
        player.$fire();
        game.delay(0.5);

        // 摸牌动画
        await player.draw(2);
        player.$draw(2);
        game.delay(0.5);

        // 伤害特效
        let target = await player.chooseTarget('选择一名目标').forResult();
        if(target.bool){
            target = target.targets[0];
            target.$damage('fire');
            await target.damage('fire');
            game.delay(0.5);
        }
    }
}
```

2. 创建一个连续动画技能：
   - 包含多个特效
   - 实现动画序列
   - 添加条件判断

<details>
<summary>参考答案 | 🟩 Easy</summary>

```javascript
"sequence_animation": {
    enable: "phaseUse",
    skillAnimation: true,
    animationColor: "thunder",
    filter: function(event, player){
        return player.countCards('h') > 0;
    },
    content: async function (event, trigger, player){
        // 弃牌动画
        let cards = await player.chooseCard('h', true).forResult();
        if(cards.bool){
            player.$throw(cards.cards);
            await player.discard(cards.cards);
            await game.delay(0.5);
        }

        // 连续特效
        player.$thunder();
        await game.delay(0.3);
        player.$fire();
        await game.delay(0.3);

        // 根据条件播放不同特效
        if(player.hp <= 2){
            // 生命值低时回复
            player.$recover();
            await player.recover();
            await game.delay(0.5);
        } else {
            // 生命值高时伤害
            let target = await player.chooseTarget('选择一名目标').forResult();
            if(target.bool){
                let target = target.targets[0];
                target.$damage('thunder');
                await target.damage('thunder');
                await game.delay(0.5);
            }
        }

        // 结束特效
        player.$fullscreenpop('技能结束', 'thunder');
        await game.delay(0.5);
    }
}
```
</details>

3. 创建一个复杂动画技能：
   - 结合多种动画类型
   - 实现动态特效
   - 优化动画性能

<details>
<summary>参考答案 | 🟨 Medium</summary>

```javascript
"complex_animation": {
    // 缓存动画状态
    init: function(player){
        player.storage.animation_state = {
            count: 0,
            lastEffect: null,
            effects: ['fire', 'thunder', 'water']
        };
    },
    
    // 动画效果管理(自建函数)
    getNextEffect: function(player){
        let state = player.storage.animation_state;
        state.count++;
        let index = state.count % state.effects.length;
        state.lastEffect = state.effects[index];
        return state.lastEffect;
    },
    
    enable: "phaseUse",
    usable: 1,
    filter: function(event, player){
        return player.countCards('h') > 0;
    },
    content: async function (event, trigger, player){
        // 批量处理动画
        game.broadcastAll(function(player){
            // 同时播放多个动画
            player.$skill('complex_animation','legend','fire');
            player.$fullscreenpop('技能发动', 'fire');
        }, player);
        await game.delay(0.8);

        // 选择目标
        let targets = await player.chooseTarget([1,3], '选择1-3个目标').forResult();
        if(targets.bool){
            event.targets = targets.targets;
            // 记录目标数量
            event.count = targets.targets.length;
        } else {
            event.finish();
        }

        // 对每个目标执行动画
        while(event.targets.length){
            let target = event.targets.shift();
            // 获取动态特效
            let effect = lib.skill.complex_animation.getNextEffect(player);
            
            // 批量播放动画
            game.broadcastAll(function(target, effect){
                target.$damage(effect);
                target.$thunder();
            }, target, effect);
            
            await target.damage(effect);
            await game.delay(0.3);
        }

        // 根据目标数量播放结束特效
        if(event.count >= 3){
            // 三目标特效
            game.broadcastAll(function(player){
                player.$thunder();
                player.$fire();
                player.$fullscreenpop('三重打击', 'fire');
            }, player);
            await game.delay(0.8);
        } else if(event.count == 2){
            // 双目标特效
            game.broadcastAll(function(player){
                player.$fire();
                player.$fullscreenpop('双重打击', 'thunder');
            }, player);
            await game.delay(0.5);
        }

        // 清理动画状态
        delete player.storage.animation_state;
    },
    ai: {
        order: 8,
        result: {
            target: -1.5
        }
    }
}
```
